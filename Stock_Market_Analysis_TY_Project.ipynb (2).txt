    "# ── Feature Engineering ───────────────────────────────────────────────────────\n",
    "# Moving Averages\n",
    "df['MA_20']  = df['Close'].rolling(window=20).mean()   # 20-day MA\n",
    "df['MA_50']  = df['Close'].rolling(window=50).mean()   # 50-day MA\n",
    "df['MA_200'] = df['Close'].rolling(window=200).mean()  # 200-day MA\n",
    "\n",
    "# Bollinger Bands\n",
    "df['BB_std']    = df['Close'].rolling(window=20).std()\n",
    "df['BB_upper']  = df['MA_20'] + (2 * df['BB_std'])\n",
    "df['BB_lower']  = df['MA_20'] - (2 * df['BB_std'])\n",
    "\n",
    "# Daily Return\n",
    "df['Daily_Return'] = df['Close'].pct_change() * 100\n",
    "\n",
    "# Volatility (30-day rolling std)\n",
    "df['Volatility'] = df['Daily_Return'].rolling(window=30).std()\n",
    "\n",
    "# RSI — Relative Strength Index\n",
    "delta     = df['Close'].diff()\n",
    "gain      = delta.clip(lower=0).rolling(14).mean()\n",
    "loss      = (-delta.clip(upper=0)).rolling(14).mean()\n",
    "rs        = gain / loss\n",
    "df['RSI'] = 100 - (100 / (1 + rs))\n",
    "\n",
    "print(\"✅ Technical indicators computed!\")\n",
    "df[['Date','Close','MA_20','MA_50','MA_200','RSI','Daily_Return','Volatility']].tail(10)"

   ]

   "outputs": [],
   "source": [
    "# ── Prepare LSTM Data ─────────────────────────────────────────────────────────\n",
    "LOOKBACK = 60  # use past 60 days to predict the next day\n",
    "\n",
    "close_prices = df['Close'].values.reshape(-1, 1)\n",
    "\n",
    "# Normalize\n",
    "scaler = MinMaxScaler(feature_range=(0, 1))\n",
    "scaled = scaler.fit_transform(close_prices)\n",
    "\n",
    "# Create sequences\n",
    "def create_sequences(data, lookback):\n",
    "    X, y = [], []\n",
    "    for i in range(lookback, len(data)):\n",
    "        X.append(data[i - lookback:i, 0])\n",
    "        y.append(data[i, 0])\n",
    "    return np.array(X), np.array(y)\n",
    "\n",
    "X_seq, y_seq = create_sequences(scaled, LOOKBACK)\n",
    "\n",
    "# Train/Test split (80/20)\n",
    "split     = int(len(X_seq) * 0.8)\n",
    "X_tr, X_te = X_seq[:split], X_seq[split:]\n",
    "y_tr, y_te = y_seq[:split], y_seq[split:]\n",
    "\n",
    "# Reshape for LSTM: (samples, timesteps, features)\n",
    "X_tr = X_tr.reshape((X_tr.shape[0], X_tr.shape[1], 1))\n",
    "X_te = X_te.reshape((X_te.shape[0], X_te.shape[1], 1))\n",
    "\n",
    "print(f\"Train shape : {X_tr.shape}\")\n",
    "print(f\"Test  shape : {X_te.shape}\")"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "# ── Build LSTM Model ──────────────────────────────────────────────────────────\n",
    "tf.random.set_seed(42)\n",
    "\n",
    "model = Sequential([\n",
    "    LSTM(64, return_sequences=True, input_shape=(LOOKBACK, 1)),\n",
    "    Dropout(0.2),\n",
    "    LSTM(64, return_sequences=False),\n",
    "    Dropout(0.2),\n",
    "    Dense(32, activation='relu'),\n",
    "    Dense(1)\n",
    "])\n",
    "\n",
    "model.compile(optimizer='adam', loss='mean_squared_error')\n",
    "model.summary()"
   ]
  },
